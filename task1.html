import matplotlib.pyplot as plt
import seaborn as sns

# Plotting the distribution of Model Year
plt.figure(figsize=(14, 6))

plt.subplot(1, 3, 1)
sns.histplot(df['Model Year'], kde=True, bins=30)
plt.title('Distribution of Model Year')

# Plotting the distribution of Electric Range
plt.subplot(1, 3, 2)
sns.histplot(df['Electric Range'].dropna(), kde=True, bins=30)
plt.title('Distribution of Electric Range')

# Plotting the distribution of Base MSRP
plt.subplot(1, 3, 3)
sns.histplot(df['Base MSRP'].dropna(), kde=True, bins=30)
plt.title('Distribution of Base MSRP')

plt.tight_layout()
plt.show()
